基本設定


// in js
import Vue from 'vue'

new Vue({
  el: '#app',
  components: {
    // ... //
  }
})

所有郵遞區號


// in js
import Zipcode from './components/zipcodes.vue'

...
  components: {
    TwzipcodeZipcode: Zipcode,
  },
...

// in HTML
// <twzipcode-county option-template=":zipcode :county-:city"></twzipcode-county>

依縣市篩選


// in js
import Zipcode from './components/zipcodes.vue'
import County from './components/counties.vue'

// https://laracasts.com/discuss/channels/vue/use-a-global-event-bus
const bus = new Vue({})

...
  components: {
    TwzipcodeZipcode: Zipcode,
    TwzipcodeCounty: County
  },
  data () {
    return {
      bus
    }
  }
...

// in HTML
// <twzipcode-county id="twzipcode__county--4" selected="臺中市"></twzipcode-county>
// <twzipcode-zipcode id="twzipcode__zipcode--4" county-id="twzipcode__county--4" init-county="臺中市" selected="423"></twzipcode-zipcode>
