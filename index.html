基本設定


// in js
import Vue from 'vue'

new Vue({
  el: '#app',
  components: {
    // ... //
  }
})

所有郵遞區號


// in js
import Zipcode from './components/zipcodes.vue'

...
  components: {
    TwzipcodeZipcode: Zipcode,
  },
...

// in HTML
// <twzipcode-county text-template=":id :county-:city"></twzipcode-county>

以縣市分組


// in js
import ZipcodeGroupby from './components/zipcodes.groupby.vue'

...
  components: {
    TwzipcodeZipcodeGroupby: ZipcodeGroupby,
  },
...

// in HTML

依縣市篩選


// in js
import Zipcode from './components/zipcodes.vue'
import County from './components/counties.vue'

// https://laracasts.com/discuss/channels/vue/use-a-global-event-bus
const bus = new Vue({})

...
  components: {
    TwzipcodeZipcode: Zipcode,
    TwzipcodeCounty: County
  },
  data () {
    return {
      bus
    }
  }
...

// in HTML
// <twzipcode-county id="twzipcode__county" selected="臺中市"></twzipcode-county>
// <twzipcode-zipcode id="twzipcode__zipcode" county-id="twzipcode__county--4" init-county="臺中市" selected="423"></twzipcode-zipcode>

樣式


// in css
.twzipcode__county {
  background-color: red;
  color: white;
}
// 如果只有一個，可換成 #twzipcode__county 

.twzipcode__zipcode--groupby {
  background-color: yellow;
  color: brown;
}
// 如果只有一個，可換成 #twzipcode__zipcode--groupby

.twzipcode__zipcode {
  background-color: lightskyblue;
  color: dimgrey;
}
// 如果只有一個，可換成 #twzipcode__zipcode
