value 和 text 可同時使用不同的 locale 。


<twzipcode-county value-locale="en"
                  text-locale="en"></twzipcode-county>

#事件

demoCallback：

若有在 parent 的 data 給予 bus 參數作為事件的 emitter ，則 components 下的每個 component 會分別 emit${id}:change事件，並送出一個更新後的值value；可使用$on來接收。


import Vue from 'vue'
import { Zipcode } from 'vue-twzipcode'

// https://laracasts.com/discuss/channels/vue/use-a-global-event-bus
const bus = new Vue({})

bus.$on('twzipcode__zipcode:change', function(event) {
  document.getElementById('demoCallback').textContent = event.value
})

new Vue({
  el: '#app',
  components: {
    TwzipcodeCounty: County
  },
  data () {
    return {
      bus
    }
  }
})

# 依縣市篩選

若只有一組縣市與郵遞區號，則無需任何更動；若有多組，則需要指定id，以便在切換縣市時不會影響到其他組的郵遞區號。 

// in js
import { Zipcode, County } from 'vue-twzipcode'

// https://laracasts.com/discuss/channels/vue/use-a-global-event-bus
const bus = new Vue({})

...
  components: {
    TwzipcodeZipcode: Zipcode,
    TwzipcodeCounty: County
  },
  data () {
    return {
      bus
    }
  }
...
          
        

<twzipcode-county id="twzipcode__county--8"
                  selected="臺中市"></twzipcode-county>

<twzipcode-zipcode id="twzipcode__zipcode--8"
                   listen-to-county="twzipcode__county--8"
                   filter-by-county="臺中市"
